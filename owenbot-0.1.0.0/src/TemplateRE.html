{-# LANGUAGE OverloadedStrings #-}

module TemplateRE
    ( trailingWS
    , accoladedArgRE
    , quotedArgRE
    , spaceRE
    ) where

import qualified Data.Text as T

trailingWS :: T.Text
trailingWS :: Text
trailingWS = Text
" *$"

quotedArgRE :: T.Text
quotedArgRE :: Text
quotedArgRE = Text
"\"(.*)\""

accoladedArgRE :: T.Text
accoladedArgRE :: Text
accoladedArgRE  = Text
"({.*})"

spaceRE :: T.Text
spaceRE :: Text
spaceRE = Text
" +"
